### Code Example: JWT Implementation
```typescript
// lib/auth/jwt.ts
import jwt from 'jsonwebtoken'

export interface JWTPayload {
  userId: string
  role: string
  churchId?: string
  email: string
  iat: number
  exp: number
}

export function generateToken(payload: Omit<JWTPayload, 'iat' | 'exp'>): string {
  return jwt.sign(payload, process.env.NEXTAUTH_SECRET!, {
    expiresIn: '15m',
    algorithm: 'HS256'
  })
}

export function verifyToken(token: string): JWTPayload | null {
  try {
    return jwt.verify(token, process.env.NEXTAUTH_SECRET!) as JWTPayload
  } catch {
    return null
  }
}

export function generateRefreshToken(userId: string): string {
  return jwt.sign({ userId }, process.env.NEXTAUTH_SECRET!, {
    expiresIn: '7d'
  })
}
```

### Example: RBAC Middleware
```typescript
// lib/middleware/requireRole.ts
export function requireRole(...roles: string[]) {
  return async (handler: ApiHandler) => {
    return async (req: NextRequest) => {
      const token = extractToken(req)
      const payload = verifyToken(token)
      
      if (!payload || !roles.includes(payload.role)) {
        return NextResponse.json(
          { error: 'Forbidden: insufficient permissions' },
          { status: 403 }
        )
      }
      
      req.user = payload
      return handler(req)
    }
  }
}

// Usage in endpoint
export const DELETE = requireRole('admin', 'moderator')(async (req) => {
  const ideaId = req.nextUrl.searchParams.get('id')
  // Admin/moderator can delete ideas
  const { error } = await supabase.from('ideas').delete().eq('id', ideaId)
  return NextResponse.json({ success: !error })
})
```

### RLS Policies Example
```sql
-- Enable RLS on all tables
ALTER TABLE communities ENABLE ROW LEVEL SECURITY;
ALTER TABLE community_threads ENABLE ROW LEVEL SECURITY;
ALTER TABLE projects ENABLE ROW LEVEL SECURITY;
ALTER TABLE ideas ENABLE ROW LEVEL SECURITY;
ALTER TABLE admin_logs ENABLE ROW LEVEL SECURITY;

-- Communities: members only
CREATE POLICY "members_can_view" ON communities
  FOR SELECT USING (
    auth.uid() IN (
      SELECT user_id FROM member_profiles
      WHERE member_profiles.community_id = communities.id
    )
    OR auth.jwt() ->> 'role' = 'admin'
  );

-- Ideas: public read, authenticated write
CREATE POLICY "public_read_ideas" ON ideas
  FOR SELECT USING (status = 'published' OR auth.uid() = author_id);

CREATE POLICY "authenticated_write_ideas" ON ideas
  FOR INSERT WITH CHECK (auth.role() = 'authenticated');

-- Admin logs: admin only
CREATE POLICY "admin_only_logs" ON admin_logs
  FOR SELECT USING (auth.jwt() ->> 'role' = 'admin');
```

## 📝 Implementation Notes

### Role Hierarchy
```
admin
  ├─ moderator (subset of admin permissions)
  ├─ church_admin (manage church + projects)
  ├─ donor (can donate + vote)
  └─ member (basic permissions)
```

### Token Lifecycle
```
1. User login → Generate JWT (15-min exp) + Refresh Token (7-day exp)
2. API requests → Include JWT in Authorization header
3. JWT expires → Client calls /api/auth/refresh with refresh token
4. New JWT issued → Client continues with new token
5. Refresh token expires → User must log in again
```

### Security Considerations
- All tokens signed with NEXTAUTH_SECRET
- Tokens never stored client-side in localStorage (use httpOnly cookies)
- Token refresh requires valid refresh token
- Failed auth attempts logged (Story 45.7 - Admin dashboard)
- Rate limiting on auth endpoints (5 attempts/min per IP)
