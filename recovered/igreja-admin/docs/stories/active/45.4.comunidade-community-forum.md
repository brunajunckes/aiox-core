### Example: Create Thread API
```typescript
// app/api/comunidade/[id]/threads/route.ts
export const POST = requireAuth(async (req: NextRequest) => {
  const { communityId } = req.params
  const { title, description, category } = await req.json()
  
  const validation = threadSchema.safeParse({ title, description, category })
  if (!validation.success) return validateError(validation.error)
  
  const { data, error } = await supabase
    .from('community_threads')
    .insert({
      community_id: communityId,
      author_id: req.user.id,
      title,
      description,
      category,
      created_at: new Date()
    })
    .select()
  
  if (error) return NextResponse.json({ error: error.message }, { status: 500 })
  return NextResponse.json(data[0], { status: 201 })
})
```

### Example: Reputation System
```typescript
// lib/reputation.ts
export async function awardReputation(
  userId: string,
  action: 'comment' | 'thread' | 'like',
  amount?: number
) {
  const points = {
    comment: 1,
    thread: 5,
    like: 2
  }[action]
  
  await supabase
    .from('member_profiles')
    .update({ reputation: reputation_increment(points) })
    .eq('user_id', userId)
}
```
